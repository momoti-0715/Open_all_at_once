exeファイルを実行すると下で設定したものがまとめて開きます
(windowsでの動作を想定)

open_all_at_once.exeとsetting.jsonは同じディレクトリ内に格納してください
パスは全て実行ファイルの絶対パスで記述(バックスラッシュはエスケープ処理してください)

※コマンド実行は自己責任で


以下はjsonの記載ルール

{
    urlを開く
    "web_open": {
        "is_open": "False", 使う場合はTureにする
        "browser": ブラウザを指定する場合はパスを記載(nullの場合はデフォルトのブラウザ),
        "url_list": [
            url1,
            url2,
            ...(指定しない場合はnull)
        ]
    },
    コマンドプロンプトを開く
    "cmd_open": {
        "is_open": "False",
        "p_c_list":[ "開きたい数だけ{}で追加する"
            {
                "path": カレントディレクトリのパス(指定しない場合はnull),
                "command": 事前にコマンドを使用する場合は記載(使用しない場合はnull) パイプは使用不可
            },
            {
                ...(さらに開きたい場合は↑に従って記述する)
            }
        ]
        
    },
    タスクを開く
    "task_open": {
        "is_open": "False",
        "task_list": [
            タスクのパス1,
            タスクのパス2,
            ...(指定しない場合はnull)
        ]
    }
}
